let libraryData = []; // 1. جلب البيانات تلقائياً من ملف JSON عند فتح الموقع لبناء الصفحة async function loadLibraryData() { try { const response = await fetch('./data.json'); libraryData = await response.json(); renderLibraryInterface(); } catch (error) { console.error("خطأ في جلب بيانات الملفات:", error); } } // 2. بناء قوائم الكتب والمقالات في واجهة الموقع ديناميكياً function renderLibraryInterface() { const categories = { 'book': document.querySelector('.category-box:nth-child(1) ul'), 'article': document.querySelector('.category-box:nth-child(2) ul'), 'worksheet': document.querySelector('.category-box:nth-child(3) ul') }; // تفريغ القوائم القديمة أولاً Object.values(categories).forEach(ul => { if(ul) ul.innerHTML = ''; }); // توزيع الملفات في مكانها الصحيح داخل الموقع تلقائياً libraryData.forEach(item => { const ul = categories[item.type]; if (ul) { const li = document.createElement('li'); li.innerHTML = `📥 ${item.title}.pdf`; ul.appendChild(li); } }); } // تشغيل الجلب التلقائي فور تحميل الصفحة window.onload = loadLibraryData; function toggleChat() { const chatContainer = document.getElementById('chatContainer'); chatContainer.style.display = chatContainer.style.display === 'flex' ? 'none' : 'flex'; } function handleKeyPress(event) { if (event.key === 'Enter') { sendMessage(); } } function sendMessage() { const inputElement = document.getElementById('chatInput'); const messageText = inputElement.value.trim(); if (!messageText) return; appendMessage(messageText, 'user-message'); inputElement.value = ''; const typingIndicator = appendMessage('جاري البحث وتوليد الإجابة...', 'bot-message'); setTimeout(() => { typingIndicator.remove(); const responseText = getSmartLocalResponse(messageText); appendMessage(responseText, 'bot-message'); }, 800); } function appendMessage(text, className) { const chatBody = document.getElementById('chatBody'); const messageElement = document.createElement('div'); messageElement.className = `message ${className}`; messageElement.innerHTML = text; chatBody.appendChild(messageElement); chatBody.scrollTop = chatBody.scrollHeight; return messageElement; } // 3. دالة ردود البوت الذكية التي تبحث داخل ملف الـ JSON تلقائياً function getSmartLocalResponse(prompt) { const query = prompt.toLowerCase(); // فحص طلبات التحميل والملفات if (query.includes('تحميل') || query.includes('كتاب') || query.includes('مقالة') || query.includes('مقال') || query.includes('ملف') || query.includes('أريد')) { // البحث في قاعدة بيانات الـ JSON عن أي ملف يطابق كلمات المستخدم let matched = libraryData.filter(item => { return query.includes(item.title.replace('كتاب ', '').replace('مقال ', '').replace('ورقة عمل ', '').toLowerCase()) || item.title.split(' ').some(word => word.length > 3 && query.includes(word)); }); let downloadReply = "بالتأكيد! إليك المواد العلمية المتاحة للتحميل المباشر بناءً على طلبك المخصص:

"; if (matched.length > 0) { matched.forEach(item => { downloadReply += `📥 تحميل ${item.title}
`; }); return downloadReply; } // عرض كافة الملفات الموجودة في الـ JSON تلقائياً إذا كان الطلب عاماً let allFilesReply = "إليك روابط التحميل السريعة لجميع المواد والملفات المتاحة في قاعدة بيانات المكتبة الآن:

"; libraryData.forEach(item => { allFilesReply += `📂 ${item.title} `; }); return allFilesReply; } // قسم الشروحات النظرية if (query.includes('مرحبا') || query.includes('أهلا') || query.includes('سلام')) { return "مرحباً بك في المكتبة الرقمية الذكية المدارة بالكامل ديناميكياً! كيف يمكنني مساعدتك في الشرح العلمي أو التحميل اليوم؟"; } if (query.includes('ديوي') || query.includes('تصنيف')) { return "تصنيف ديوي العشري (DDC): هو نظام لتصنيف محتويات المكتبات وضعه ميلفيل ديوي عام 1876. يعتمد النظام على تقسيم المعرفة البشرية إلى 10 أصول رئيسية."; } if (query.includes('فهرسة') || query.includes('الفهرس')) { return "الفهرسة الإلكترونية: هي عملية الوصف الفني والموضوعي لمصادر المعلومات باستخدام الحاسب الآلي، لإنتاج فهارس متاحة عبر الإنترنت تسهل الوصول للكتب بسرعة."; } if (query.includes('ذكاء') || query.includes('تعليم')) { return "الذكاء الاصطناعي في التعليم والمكتبات: يساهم في تخصيص التعلم للطلاب، وفي المكتبات يساعد في أتمتة عمليات التصنيف، وتحسين محركات البحث الموضوعي."; } return "شكرًا لسؤالك. يمكنني شرح المفاهيم العلمية أو تزويدك بروابط تحميل أي ملف متوفر في قاعدة البيانات، فقط اكتب لي اسماً واضحاً للملف الذي تبحث عنه."; }
